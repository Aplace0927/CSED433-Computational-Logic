\begin{lemma}
  $((A \lor B) \supset C) \supset ((A \supset C) \land (B \supset C))$ $true$
\end{lemma}

\begin{center}
  \begin{prooftree}
    \infer0[$x$]{A \lor B \supset C \ true}
    \infer0[$a$]{A \ true}
    \infer1[\disjIL]{A \lor B \ true}
    \infer2[\implE]{C \ true}
    \infer1[\implI{a}]{A \supset C \ true}
    \infer0[$x$]{A \lor B \supset C \ true}
    \infer0[$b$]{B \ true}
    \infer1[\disjIL]{A \lor B \ true}
    \infer2[\implE]{C \ true}
    \infer1[\implI{b}]{B \supset C \ true}
    \infer2[\conjI]{(A \supset B) \land (A \supset C) \ true}
    \infer1[\implI{x}]{((A \lor B) \supset C) \supset ((A \supset C) \land (B \supset C)) \ true}
  \end{prooftree}
\end{center}

    \infer1[\conjEL]{A \supset C \ true}

    \infer2[\implE]{C \ true}

\section{Problem 4}
\begin{lemma}
  $(\lnot\lnot(A \lor \lnot A) \supset (A \lor \lnot A)) \supset (A \lor \lnot A)$ $true$
\end{lemma}
\begin{center}
  \begin{prooftree}
    \infer0[$f$]{\lnot\lnot(A \lor \lnot A) \supset (A \lor \lnot A) \ true}
    \hypo{A \lor \lnot A \ true}
    \infer0[$a$]{A \ true}
    \infer0[$na$]{\lnot A \ true}
    \infer3[\disjE{a}{na}]{A \lor \lnot A \ true}
    \infer0[$an$]{\lnot(A \lor \lnot A) \ true}
    \infer2[\negE]{\bot \ true}
    \infer1[\negI{an}]{\lnot\lnot (A \lor \lnot A) \ true}
    \infer2[\implE]{A \lor \lnot A \ true}
    \infer1[\implI{f}]{(\lnot\lnot(A \lor \lnot A) \supset (A \lor \lnot A)) \supset (A \lor \lnot A) \ true}

  \end{prooftree}
\end{center}
